1""" 

2Use this plugin to activate coverage report. 

3 

4To use this plugin, you need to install ``nose2[coverage_plugin]``. e.g. 

5 

6:: 

7 

8 $ pip install nose2[coverage_plugin]>=0.6.5 

9 

10 

11Then, you can enable coverage reporting with : 

12 

13:: 

14 

15 $ nose2 --with-coverage 

16 

17Or with this lines in ``unittest.cfg`` : 

18 

19:: 

20 

21 [coverage] 

22 always-on = True 

23 

24 

25You can further specify coverage behaviors with a ``.coveragerc`` file, as 

26specified by `Coverage Config 

27<http://coverage.readthedocs.io/en/latest/config.html>`_. 

28However, when doing so you should also be aware of 

29`Differences From coverage`_.

104 def beforeSummaryReport(self, event): 

105 """Only called if active so stop coverage and produce reports.""" 

106 if self.covController: 

107 self.covController.stop() 

108 

109 # write to .coverage file 

110 # do this explicitly (instead of passing auto_data=True to 

111 # Coverage constructor) in order to not load an existing .coverage 

112 # this better imitates the behavior of invoking `coverage` from the 

113 # command-line, which sets `Coverage._auto_save` (triggers atexit 

114 # saving to this file), but not `Coverage._auto_load` 

115 # requesting a better fix in nedbat/coveragepy#34 

116 self.covController.save() 

117 

118 if self._mpmode: 

119 self.covController.combine(strict=True) 

120 

121 percent_coverage = None 

122 

123 if 'term' in self.covReport or 'term-missing' in self.covReport: 

124 # only pass `show_missing` if "term-missing" was given 

125 # otherwise, just allow coverage to load show_missing from 

126 # config 

127 kwargs = {} 

128 if 'term-missing' in self.covReport: 

129 kwargs['show_missing'] = True 

130 percent_coverage = self.covController.report( 

131 file=self.error_output_buffer, **kwargs) 

132 if 'annotate' in self.covReport: 

133 percent_coverage = self.covController.annotate() 

134 if 'html' in self.covReport: 

135 percent_coverage = self.covController.html_report() 

136 if 'xml' in self.covReport: 

137 percent_coverage = self.covController.xml_report() 

138 

139 fail_under = self.covController.get_option("report:fail_under") 

140 if (fail_under is not None and percent_coverage is not None and 

141 fail_under > percent_coverage): 

142 self.decided_failure = True 

143

145 self._mpmode = True 

146 self._subprocess = True 

147 self._start_coverage() 

148 

149 def stopSubprocess(self, event): 

150 self.covController.stop()

152 

153 def wasSuccessful(self, event): 

154 """Mark full test run as successful or unsuccessful""" 

155 if self.decided_failure: 

156 event.success = False 

157 

158 def afterSummaryReport(self, event): 

159 """Reporting data is collected, failure status determined and set. 

160 Now print any buffered error output saved from beforeSummaryReport""" 

161 print(self.error_output_buffer.getvalue(), file=event.stream)

164 try: 

165 import coverage 

166 except ImportError: 

167 print('Warning: you need to install "coverage_plugin" ' 

168 'extra requirements to use this plugin. ' 

169 'e.g. `pip install nose2[coverage_plugin]`')

172 self.covController = coverage.Coverage(
