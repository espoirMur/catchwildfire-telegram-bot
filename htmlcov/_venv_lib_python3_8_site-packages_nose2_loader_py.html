19class PluggableTestLoader(object): 

20 

21 """Test loader that defers all loading to plugins 

22 

23 :param session: Test run session. 

24 

25 .. attribute :: suiteClass 

26 

27 Suite class to use. Default: :class:`unittest.TestSuite`. 

28 

29 """ 

30 suiteClass = unittest.TestSuite

32 def __init__(self, session): 

33 self.session = session

35 def loadTestsFromModule(self, module): 

36 """Load tests from module. 

37 

38 Fires :func:`loadTestsFromModule` hook. 

39 

40 """ 

41 evt = events.LoadFromModuleEvent(self, module) 

42 result = self.session.hooks.loadTestsFromModule(evt) 

43 if evt.handled: 

44 suite = result or self.suiteClass() 

45 else: 

46 suite = self.suiteClass(evt.extraTests) 

47 filterevt = events.ModuleSuiteEvent(self, module, suite) 

48 result = self.session.hooks.moduleLoadedSuite(filterevt) 

49 if result: 

50 return result or self.suiteClass() 

51 return filterevt.suite 

52 

53 def loadTestsFromNames(self, testNames, module=None): 

54 """Load tests from test names. 

55 

56 Fires :func:`loadTestsFromNames` hook. 

57 

58 """ 

59 event = events.LoadFromNamesEvent( 

60 self, testNames, module) 

61 result = self.session.hooks.loadTestsFromNames(event) 

62 log.debug('loadTestsFromNames event %s result %s', event, result) 

63 if event.handled: 

64 suites = result or [] 

65 else: 

66 if event.names: 

67 suites = [self.loadTestsFromName(name, module) 

68 for name in event.names] 

69 elif module: 

70 suites = self.loadTestsFromModule(module) 

71 if event.extraTests: 

72 suites.extend(event.extraTests) 

73 return self.suiteClass(suites) 

74 

75 def loadTestsFromName(self, name, module=None): 

76 """Load tests from test name. 

77 

78 Fires :func:`loadTestsFromName` hook. 

79 

80 """ 

81 log.debug('loadTestsFromName %s/%s', name, module) 

82 event = events.LoadFromNameEvent(self, name, module) 

83 result = self.session.hooks.loadTestsFromName(event) 

84 if event.handled: 

85 suite = result or self.suiteClass() 

86 return suite 

87 return self.suiteClass(event.extraTests)

89 def failedImport(self, name): 

90 """Make test case representing a failed import.""" 

91 message = 'Failed to import test module: %s' % name 

92 if hasattr(traceback, 'format_exc'): 

93 # Python 2.3 compatibility 

94 # format_exc returns two frames of discover.py as well XXX ? 

95 message += '\n%s' % traceback.format_exc() 

96 return self._makeFailedTest( 

97 'ModuleImportFailure', name, ImportError(message))

99 def failedLoadTests(self, name, exception): 

100 """Make test case representing a failed test load.""" 

101 return self._makeFailedTest('LoadTestsFailure', name, exception)

107 def discover(self, start_dir=None, pattern=None): 

108 """Compatibility shim for ``load_tests`` protocol.""" 

109 try: 

110 oldsd = self.session.startDir 

111 self.session.startDir = start_dir 

112 return self.loadTestsFromNames([]) 

113 finally: 

114 self.session.startDir = oldsd

117 def testFailure(self): 

118 if isinstance(exception, Exception): 

119 raise exception

121 # exception tuple (type, value, traceback) 

122 six.reraise(*exception) 

123 attrs = {methodname: testFailure}
