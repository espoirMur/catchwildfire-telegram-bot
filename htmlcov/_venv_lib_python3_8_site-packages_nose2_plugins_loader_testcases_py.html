29class TestCaseLoader(events.Plugin): 

30 

31 """Loader plugin that loads from test cases""" 

32 alwaysOn = True 

33 configSection = 'testcases' 

34 

35 def registerInSubprocess(self, event): 

36 event.pluginClasses.append(self.__class__)

38 def loadTestsFromModule(self, event): 

39 """Load tests in :class:`unittest.TestCase` subclasses""" 

40 seen = set() 

41 module = event.module 

42 for name in dir(module): 

43 obj = getattr(module, name) 

44 if id(obj) in seen: 

45 continue 

46 seen.add(id(obj)) 

47 if isinstance(obj, type) and issubclass(obj, unittest.TestCase): 

48 event.extraTests.append( 

49 self._loadTestsFromTestCase(event, obj)) 

50 

51 def loadTestsFromName(self, event): 

52 """Load tests from event.name if it names a test case/method""" 

53 name = event.name 

54 module = event.module 

55 log.debug("load %s from %s", name, module) 

56 try: 

57 result = util.test_from_name(name, module) 

58 except (AttributeError, ImportError): 

59 event.handled = True 

60 return event.loader.failedLoadTests(name, sys.exc_info()) 

61 if result is None: 

62 return 

63 parent, obj, name, index = result 

64 if isinstance(obj, type) and issubclass(obj, unittest.TestCase): 

65 # name is a test case class 

66 event.extraTests.append(self._loadTestsFromTestCase(event, obj)) 

67 elif (isinstance(parent, type) and 

68 issubclass(parent, unittest.TestCase) and 

69 not util.isgenerator(obj) and 

70 not hasattr(obj, 'paramList')): 

71 # name is a single test method 

72 event.extraTests.append(parent(obj.__name__))

74 def _loadTestsFromTestCase(self, event, testCaseClass): 

75 evt = events.LoadFromTestCaseEvent(event.loader, testCaseClass) 

76 result = self.session.hooks.loadTestsFromTestCase(evt) 

77 if evt.handled: 

78 loaded_suite = result or event.loader.suiteClass() 

79 else: 

80 names = self._getTestCaseNames(event, testCaseClass) 

81 if not names and hasattr(testCaseClass, 'runTest'): 

82 names = ['runTest'] 

83 # FIXME return failure test case if name not in testcase class 

84 loaded_suite = event.loader.suiteClass(map(testCaseClass, names)) 

85 if evt.extraTests: 

86 loaded_suite.addTests(evt.extraTests) 

87 return loaded_suite

89 def _getTestCaseNames(self, event, testCaseClass): 

90 excluded = set() 

91 

92 def isTestMethod(attrname, testCaseClass=testCaseClass, 

93 excluded=excluded): 

94 prefix = evt.testMethodPrefix or self.session.testMethodPrefix 

95 return ( 

96 attrname.startswith(prefix) and 

97 hasattr(getattr(testCaseClass, attrname), '__call__') and 

98 attrname not in excluded 

99 ) 

100 evt = events.GetTestCaseNamesEvent( 

101 event.loader, testCaseClass, isTestMethod) 

102 result = self.session.hooks.getTestCaseNames(evt) 

103 if evt.handled: 

104 test_names = result or [] 

105 else: 

106 excluded.update(evt.excludedNames) 

107 test_names = [entry for entry in dir(testCaseClass) 

108 if isTestMethod(entry)] 

109 if evt.extraNames: 

110 test_names.extend(evt.extraNames) 

111 sortkey = getattr( 

112 testCaseClass, 'sortTestMethodsUsing', 

113 event.loader.sortTestMethodsUsing) 

114 if sortkey: 

115 test_names.sort( 

116 key=sortkey) 

117 return test_names
